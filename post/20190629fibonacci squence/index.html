Fibonacci squence

2019-06-29 python 
def fib(n):
    """This is documentation string for function. It'll be available by fib.__doc__() 
    Return a list containing the Fibonacci series up to n."""  
    result = []
    a = 1
    b = 1
    while a < n:
        result.append(a)
        tmp_var = b
        b += a
        a = tmp_var
    return result

print(fib(10))
